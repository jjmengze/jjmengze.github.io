	basicMetricLabels := metrics.Labels{

		"probe_type": w.probeType.String(),

    //Ë®≠ÂÆö worker proberResultsSuccessfulMetricLabels

	w.proberResultsSuccessfulMetricLabels = deepCopyPrometheusLabels(basicMetricLabels)
	w.proberResultsSuccessfulMetricLabels["result"] = probeResultSuccessful

    //ÈÄèÈÅé deepCopyPrometheusLabels Êää basicMetricLabels Ë§áË£Ω‰∏Ä‰ªΩ‰∏¶‰∏î	
    //Ë®≠ÂÆö worker proberResultsFailedMetricLabels

	w.proberResultsFailedMetricLabels = deepCopyPrometheusLabels(basicMetricLabels)

    //Ë®≠ÂÆö worker proberResultsUnknownMetricLabels

	w.proberResultsUnknownMetricLabels = deepCopyPrometheusLabels(basicMetricLabels)
	w.proberResultsUnknownMetricLabels["result"] = probeResultUnknown

	//ÂõûÂÇ≥worker
	return w
}

// run periodically probes the container.
func (w *worker) run() {
	//ÈÄèÈÅé pod spec Ë®≠ÂÆö probe ÁöÑÈñìÈöîÊôÇÈñì 
	probeTickerPeriod := time.Duration(w.spec.PeriodSeconds) * time.Second

	//ÊàëÁåúÁöÑ...‰∏çÁ¢∫ÂÆöÁÇ∫‰ªÄÈ∫ºÊúÉÈÄôÊ®£Ë®≠Ë®àÔºåÂ¶ÇÊûúÊúâÁü•ÈÅìÁöÑÂ§ßÂ§ßÂ∏åÊúõËÉΩ‰∏çÂêùÂóáÂëäÁü•
	//‰æùÁÖßÁèæÂú®ÁöÑÊôÇÈñìÈªûÊ∏õÂéª kubelet ÁöÑÂïüÂãïÊôÇÈñìÔºåÂæóÂà∞ kubelet Â≠òÊ¥ªÁöÑÊôÇÈñì„ÄÇ
	//Â¶ÇÊûú PeriodSeconds Â§ßÊñº kubelet Â≠òÊ¥ªÁöÑÊôÇÈñìÁöÑË©±ÔºåËÆìworker Áù°‰∏Ä‰∏ãÁ≠âÁ≠âÂú® probe „ÄÇ
	//ÊàëÁåúÂèØËÉΩÊòØÂõ†ÁÇ∫kubelet ÈÇÑÊ≤íÂÆåÂÖ®Ê∫ñÂÇôÂ•ΩÔºàÂêßÔºüÔºâ
	if probeTickerPeriod > time.Since(w.probeManager.start) {
		time.Sleep(time.Duration(rand.Float64() * float64(probeTickerPeriod)))
	}

	//Ë®≠ÂÆö ticker Â§ö‰πÖË¶ÅËß∏Áôº‰∏ÄÊ¨°
	probeTicker := time.NewTicker(probeTickerPeriod)

	//clean up function 
	defer func() {
		// ÈóúÈñâticker
		probeTicker.Stop()
		//Â¶ÇÊûú container id ÈÇÑÂú®ÁöÑË©±Â∞±ÁßªÈô§ resultsManager Ë∑üÈÄôÂÄã container ÊúâÈóúÁöÑË≥áÊñô
		if !w.containerID.IsEmpty() {
			w.resultsManager.Remove(w.containerID)
		}
		
		//ÁßªÈô§map‰∏≠Á¥ÄÈåÑÁöÑ worker Ë≥áË®ä
		w.probeManager.removeWorker(w.pod.UID, w.container.Name, w.probeType)
		//metric ‰∏çÂú®Ë®òÈåÑÈÄôÂÄã worker ÊâÄÁôºÁîüÁöÑ metric 
		ProberResults.Delete(w.proberResultsSuccessfulMetricLabels)
		ProberResults.Delete(w.proberResultsFailedMetricLabels)
		ProberResults.Delete(w.proberResultsUnknownMetricLabels)
	}()

probeLoop:
	//ÊØèÊ¨° prob ÂÆåÊàêÂæåÂ¶ÇÊûú probe ÁµêÊûúÁÇ∫ true ÔºåÈúÄË¶ÅÁ≠âÂæÖ probeTicker ÔºåÊàñÊòØ manualTriggerCh ÂÜçÈÄ≤Ë°å‰∏ã‰∏ÄÊ¨°ÁöÑ probe trigger

		// Wait for next probe tick.
		select {
		case <-w.stopCh:
			break probeLoop
		case <-probeTicker.C:
		case <-w.manualTriggerCh:
			// continue
		}
	}
}

// probe container ‰∏¶‰∏îÂõûÂÇ≥ probe ÁµêÊûú„ÄÇÂ¶ÇÊûú probe ÈÅéÁ®ã‰∏≠ÊúâÈåØÊúÉÂõûÂÇ≥ false ÂëºÂè´ËÄÖÈúÄË¶ÅÈóúÈñâ worker„ÄÇ
func (w *worker) doProbe() (keepGoing bool) {

	defer func() { recover() }() 
	// runtime.HandleCrash Á¥ÄÈåÑ‰∏Ä‰∏ãËÄåÂ∑≤„ÄÇ
	defer runtime.HandleCrash(func(_ interface{}) { keepGoing = true })

	//ÈÄèÈÅé statusManager Ë∑üËëó UID ÊãøÂà∞ pod statusÔºàstatusManager Âú®ÈÄôË£°‰∏çÂ§™ÈáçË¶ÅÔºåÁü•ÈÅìÂèØ‰ª•ÊãøÂà∞ pod status Â∞±Â•Ω‰∫ÜÔºâ

		klog.V(3).InfoS("No status for pod", "pod", klog.KObj(w.pod))
		return true

	if status.Phase == v1.PodFailed || status.Phase == v1.PodSucceeded {
		klog.V(3).InfoS("Pod is terminated, exiting probe worker",
			"pod", klog.KObj(w.pod), "phase", status.Phase)
		return false
	}

	//ÈÅûËø¥ container status ÔºåÂà§Êñ∑ÊâÄÊúâÁöÑ container status Â∞çÊáâÂà∞Ëº∏ÂÖ•ÁöÑ ccontainer name ÂõûÂÇ≥ status ÁãÄÊÖã
	//Â¶ÇÊûúÊâæ‰∏çÂà∞Â∞çÊáâÁöÑ container status Â∞±Á≠âÂæÖ‰∏ã‰∏ÄËº™
	c, ok := podutil.GetContainerStatus(status.ContainerStatuses, w.container.Name)
	if !ok || len(c.ContainerID) == 0 {
		// ÂÆπÂô®Â∞öÊú™ÂâµÂª∫ÔºåÊàñËÄÖÂ∑≤Ë¢´Âà™Èô§„ÄÇ
		klog.V(3).InfoS("Probe target container not found",
			"pod", klog.KObj(w.pod), "containerName", w.container.Name)
		return true // Wait for more information.
	}

	//Âà§Êñ∑ worker Ë≤†Ë≤¨ÁöÑ container id ÊòØ‰∏çÊòØË∑ü status ÁöÑ container ÂèØ‰ª•Â∞ç‰∏äÔºåÂ¶ÇÊûúÂ∞ç‰∏ç‰∏äÈÄôÁ®ÆÁãÄÊ≥ÅÂèØËÉΩÁôºÁîüÂú® container Ë¢´Âà™Èô§ÊàñÊòØ container ÊîπËÆä‰∫Ü„ÄÇ
	if w.containerID.String() != c.ContainerID {
		//Â¶ÇÊûú container id ‰∏çÊòØÁ©∫ÁöÑ    
		if !w.containerID.IsEmpty() {
			//Âæû resultsManager Âà™Èô§ÈóúÊñº worker container ÁöÑÁµêÊûú        
			w.resultsManager.Remove(w.containerID)
		}
		//Ë®≠ÂÆöÊñ∞ÁöÑ container id         
		w.containerID = kubecontainer.ParseContainerID(c.ContainerID)
        //Ë®≠ÂÆö resultsManager Ë¶ÅÊé•Êî∂Êñ∞ÁöÑ worker container id Ôºå‰∏¶‰∏îÁµ¶‰ªñÂàùÂßãÂåñÁöÑ value Ëàá pod  spec ÔºàÈÄôË£°Ë¶ÅÊ≥®ÊÑèÊ†πÊìö probe ÂΩ¢Âºè‰∏çÂêå‰ªñÂÄëÂàùÂßãÂåñÂæóÊï∏ÂÄº‰πü‰∏ç‰∏ÄÊ®£ ‰æãÂ¶Çreadiness :faill  liveness :Success  startup :UnknownÔºâ
        w.resultsManager.Set(w.containerID, w.initialValue, w.pod)
		// Âõ†ÁÇ∫ÊúâÊñ∞ÁöÑ container ÊàëÂÄëÁπºÁ∫å prob ÊµÅÁ®ã
		w.onHold = false
	}

	//Âà§Êñ∑ÊòØÂê¶Ë¶ÅÁπºÁ∫å probe ÊµÅÁ®ã
	if w.onHold {
		// Worker is on hold until there is a new container.
		return true
	}

	//Âà§Êñ∑ container ÁöÑÁãÄÊÖãÊòØÂê¶Ê≠£Âú®RunningÔºåÂ¶ÇÊûú‰∏çÊòØÊ≠£Âú® Running ÊúâÂèØËÉΩÂú®ÂÅöwaiting „ÄÅÊúâÂèØËÉΩÂú® Terminated „ÄÇ
	if c.State.Running == nil {
		klog.V(3).InfoS("Non-running container probed",
			"pod", klog.KObj(w.pod), "containerName", w.container.Name)
		//Â¶ÇÊûú container id ‰∏çÊòØÁ©∫ÁöÑ   
		if !w.containerID.IsEmpty() {
			//Âæû resultsManager Âà™Èô§ÈóúÊñº worker container ÁöÑÁµêÊûú        
			w.resultsManager.Set(w.containerID, results.Failure, w.pod)
		}
		// Â¶ÇÊûú RestartPolicy ÁÇ∫ ‰∏çÈáçÊñ∞ÂïüÂãïÔºåÂâá‰∏≠Ê≠¢ worker„ÄÇ
		return c.State.Terminated == nil ||
			w.pod.Spec.RestartPolicy != v1.RestartPolicyNever
	}


	// ÈÄôÈÇäÊàëÂÄëË¶ÅÂÖà‰∫ÜËß£‰∏ÄÈªûÔºÅÔºÅÔºÅÈùûÈï∑ÈáçË¶ÅÂ¶ÇÊûú‰∏ç‰∫ÜËß£Âª∫Ë≠∞ÂÖàÁúã‰πãÂâçÊàëÈÄôÁØáÊñáÁ´†[Â≠∏ÁøíKubernetes Garbage CollectionÊ©üÂà∂](https://blog.jjmengze.website/posts/kubernetes/kubernetes-garbage-collection/)
	// Á∞°ÂñÆ‰æÜË™™Ôºåpod Ë¢´Âà™ÊéâÊúâÂèØËÉΩÂÖàÂá∫Áèæ DeletionTimestamp ÁöÑÁãÄÊÖã
	// Âú®ÈÄôÂÄãÁãÄÊÖã‰πã‰∏ã pod ÊúÉÂú® BackGround ÁãÄÊÖãË¢´ÂõûÊî∂
	
    
	// ÂèØ‰ª•ÊääÂÆÉÊÉ≥ÂÉèÊàêËôïÊñº Deletion ÁãÄÊÖãÁöÑ pod Ôºå‰∏îÊúâË®≠ÂÆö probe liveness ÊàñÊòØ startupÔºåÈÄèÈÅé resultsManager Ë®≠ÂÆöÊàê probe ÊàêÂäüÔºå‰∏çÁÑ∂ÊúÉÊää container Âà™Êéâ(ÈáçÂïü)„ÄÇ
	// ÊúÄÂæåÂÅúÊ≠¢ worker ÔºåÂõ†ÁÇ∫ pod Â∑≤Á∂ìË¶ÅË¢´Âà™Êéâ‰∫Ü worker Â∞±Ê≤íÁî®ËôïÂõâ„ÄÇ
	if w.pod.ObjectMeta.DeletionTimestamp != nil && (w.probeType == liveness || w.probeType == startup) {
		klog.V(3).InfoS("Pod deletion requested, setting probe result to success",
			"probeType", w.probeType, "pod", klog.KObj(w.pod), "containerName", w.container.Name)
		if w.probeType == startup {
			klog.InfoS("Pod deletion requested before container has fully started",
				"pod", klog.KObj(w.pod), "containerName", w.container.Name)
		}
		// Set a last result to ensure quiet shutdown.
		w.resultsManager.Set(w.containerID, results.Success, w.pod)
		// Stop probing at this point.
		return false
	}
    
    

	// Âà§Êñ∑ probe ÁöÑÂàùÂßã Delay ÊôÇÈñìÊòØÂê¶Âà∞‰∫ÜÔºåÂ¶ÇÊûúÈÇÑÊ≤íÂà∞Â∞±ÈúÄË¶ÅÁ≠â‰∏ã‰∏ÄÊ¨°Ëß∏Áôº
	if int32(time.Since(c.State.Running.StartedAt.Time).Seconds()) < w.spec.InitialDelaySeconds {
		return true
	}

	//Âà§Êñ∑ (ÈÅéÂéª) startup probe ÊòØÂê¶Â∑≤Á∂ìÊàêÂäüÔºåÂ¶ÇÊûúÂ∑≤Á∂ìÊàêÂäüÂ∞±ÂèØ‰ª•ÈóúÈñâ startup workerÔºåÂÖ∂‰ªñÁ®ÆÈ°ûÁöÑ woker ‰øùÁïô„ÄÇ
	//Â¶ÇÊûúÊòØÈÇÑÊ≤í startup probe Â§±Êïó ÂÖ∂‰ªñ probe ÈÉΩ‰∏çÁî®Ë´áÁõ¥Êé•ÈÄÄÂõûÂéªÈáçÊñ∞Á≠âÂæÖ‰∏ã‰∏ÄÊ¨°Ëß∏Áôº„ÄÇ
	if c.Started != nil && *c.Started {
		// Stop probing for startup once container has started.
		if w.probeType == startup {
			return false
		}
	} else {
		// Disable other probes until container has started.
		if w.probeType != startup {
			return true
		}
	}

	
	//ÂØ¶ÈöõÂü∑Ë°åÂêÑÁ®Æ probe ÁöÑÂú∞ÊñπÔºåÂ¶ÇÊûú probe Êúâ error Áõ¥Êé•ÂÅúÊ≠¢ worker 
	//ÈÄôË£°‰æùË≥¥‰πãÂâçÊ≥®ÂÖ•ÁöÑ probeManager ÁöÑ probe ÂØ¶‰ΩúÔºåËóâÁî±ÊàëÂÄë‰∏üÂÖ•ÁöÑ probe ÂûãÊÖãÊ±∫ÂÆöË¶ÅÊÄéÈ∫ºÂü∑Ë°å probe „ÄÇ
	result, err := w.probeManager.prober.probe(w.probeType, w.pod, status, w.container, w.containerID)
	if err != nil {
		// Prober error, throw away the result.
		return true
	}

	//Â¶ÇÊûú probe Ê≤íÊúâ error ÈÄèÈÅé ProberResults.With ÂéªËß∏Áôº metric ‰ª•Êèê‰æõÂæåÁ∫åÁõ£ÊéßÊúçÂãô
	switch result {

	case results.Failure:
		ProberResults.With(w.proberResultsFailedMetricLabels).Inc()

	if w.lastResult == result {
		w.resultRun++
	} else {

	}

	if (result == results.Failure && w.resultRun < int(w.spec.FailureThreshold)) ||

		return true
